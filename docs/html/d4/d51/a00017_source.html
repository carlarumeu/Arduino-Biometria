33T* alReves(T* p, int n) {
34 T aux;
35
36 for (int i = 0; i < n / 2; i++) {
37 aux = p[i];
38 p[i] = p[n - i - 1];
39 p[n - i - 1] = aux;
40 }
41 return p;
42} // ()
